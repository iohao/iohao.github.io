Nginx

tip

dnf 和 yum 都是用于基于 RPM 的 Linux 发行版（如 Red Hat Enterprise Linux (RHEL)、CentOS、Fedora、以及你使用的 Alibaba Cloud Linux）的软件包管理器。 简单来说，dnf 是 yum 的下一代，它旨在改进 yum 存在的不足。

安装​

更新包列表

sudo dnf update -y
sudo dnf install nginx -y

启动​

sudo systemctl start nginx

设置开机自启​

sudo systemctl enable nginx

Created symlink /etc/systemd/system/multi-user.target.wants/nginx.service → /usr/lib/systemd/system/nginx.service.

防火墙​

启动 FirewallD 服务​

sudo systemctl start firewalld

设置 FirewallD 开机自启 (推荐)​

sudo systemctl enable firewalld

Created symlink /etc/systemd/system/dbus-org.fedoraproject.FirewallD1.service → /usr/lib/systemd/system/firewalld.service.

Created symlink /etc/systemd/system/multi-user.target.wants/firewalld.service → /usr/lib/systemd/system/firewalld.service.

检查 FirewallD 状态 (可选，但推荐验证)​

你可以检查防火墙服务的状态，确认它是否正在运行

sudo systemctl status firewalld

配置防火墙​

如果你的 Alibaba Cloud Linux 服务器启用了防火墙（这是默认且推荐的安全设置），你需要开放 HTTP (80 端口) 和 HTTPS (443 端口) 流量，以便外部用户可以访问你的 Nginx 网站。

sudo firewall-cmd --permanent --add-service=http
sudo firewall-cmd --permanent --add-service=https
sudo firewall-cmd --reload
  • sudo firewall-cmd --permanent --add-service=http: 允许 HTTP (80 端口) 流量通过防火墙，--permanent 表示永久生效。
  • sudo firewall-cmd --permanent --add-service=https: 允许 HTTPS (443 端口) 流量通过防火墙，--permanent 表示永久生效。
  • sudo firewall-cmd --reload: 重新加载防火墙规则，使更改立即生效。

验证 Nginx 安装​

curl http://localhost

Nginx 的默认配置路径​

Nginx 的默认配置路径：

  • 主配置文件： /etc/nginx/nginx.conf
  • 站点配置目录： /etc/nginx/conf.d/ (通常建议在这个目录下创建独立的 .conf 文件来配置你的网站)
  • 默认网站根目录： /usr/share/nginx/html (Nginx 默认页面的文件通常存放在这里)
  • 日志文件目录： /var/log/nginx/

现在 Nginx 已经安装并运行了，你可以根据需要修改配置文件来托管你的网站或应用程序。

重启​

sudo systemctl restart nginx

检查 Nginx 状态​

检查 Nginx 服务的运行状态，确认它是否正在运行

sudo systemctl status nginx

检查配置文件语法​

sudo nginx -t

重新加载 Nginx 配置​

sudo nginx -s reload

配置文件相关​

将默认的 nginx 配置文件更改为，see include

# For more information on configuration, see:
#   * Official English Documentation: http://nginx.org/en/docs/
#   * Official Russian Documentation: http://nginx.org/ru/docs/

user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log;
pid /run/nginx.pid;

# Load dynamic modules. See /usr/share/doc/nginx/README.dynamic.
include /usr/share/nginx/modules/*.conf;

events {
    worker_connections 1024;
}

http {
    include       mime.types;
    default_type  application/octet-stream;

    # 自定义的 .conf 文件目录
    include ~/config/nginx_config/*.conf;

    sendfile        on;
    #tcp_nopush     on;
    keepalive_timeout  65;
    #gzip  on;
}

在 /etc/nginx/nginx.conf 文件中，已经配置了 include 目录了。 之后，只需要在 config/nginx_config 目录下新增相关配置即可。

创建 nginx 配置目录，用于存放自定义配置

mkdir -p config/nginx_config

自定义 nginx 配置示例​

nginx web​

创建自定义 nginx 配置文件

touch web_config.conf

输入以下内容

    server {
        listen       80;
        listen       [::]:80;
        server_name  _;
        root         /usr/share/nginx/html;

        # Load configuration files for the default server block.
        include /etc/nginx/default.d/*.conf;

        location / {
            proxy_pass http://127.0.0.1:8082;
        }

        error_page 404 /404.html;
            location = /40x.html {
        }

        error_page 500 502 503 504 /50x.html;
            location = /50x.html {
        }
    }

nginx websocket​

创建自定义 nginx 配置文件

touch websocket_config.conf
server {
    listen 10099;
    server_name localhost;

    location /websocket {
        proxy_pass http://127.0.0.1:10100;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";

    	# 将客户端的 Host 和 IP 信息一并转发到对应节点
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $host;
    }
}

ssl https​

申请证书​

证书地址 https://httpsok.com/

将证书放到 /etc/nginx/cert 目录下

完整配置​

# For more information on configuration, see:
#   * Official English Documentation: http://nginx.org/en/docs/
#   * Official Russian Documentation: http://nginx.org/ru/docs/

user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log;
pid /run/nginx.pid;

# Load dynamic modules. See /usr/share/doc/nginx/README.dynamic.
include /usr/share/nginx/modules/*.conf;

events {
    worker_connections 1024;
}

http {
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile            on;
    tcp_nopush          on;
    tcp_nodelay         on;
    keepalive_timeout   65;
    types_hash_max_size 4096;

    include             /etc/nginx/mime.types;
    default_type        application/octet-stream;

    # Load modular configuration files from the /etc/nginx/conf.d directory.
    # See http://nginx.org/en/docs/ngx_core_module.html#include
    # for more information.
    include /etc/nginx/conf.d/*.conf;

    server {
        listen       80;
        listen       [::]:80;
        server_name  _;
        root         /usr/share/nginx/html;

        # Load configuration files for the default server block.
        include /etc/nginx/default.d/*.conf;
        #将所有HTTP请求通过rewrite指令重定向到HTTPS。
        rewrite ^(.*)$ https://$host$1;

        error_page 404 /404.html;
            location = /40x.html {
        }

        error_page 500 502 503 504 /50x.html;
            location = /50x.html {
        }
    }

    server {
        listen       443 ssl http2;
        listen       [::]:443 ssl http2;
        server_name  _.iohao.com;
        root         /usr/share/nginx/html;

        # 填写证书、证书私钥文件的绝对路径
        ssl_certificate "/etc/nginx/cert/_.iohao.com.pem";
        ssl_certificate_key "/etc/nginx/cert/_.iohao.com.key";

        ssl_session_cache shared:SSL:1m;
        ssl_session_timeout  10m;
        ssl_ciphers PROFILE=SYSTEM;
        #表示优先使用服务端加密套件。默认开启
        ssl_prefer_server_ciphers on;

        # Load configuration files for the default server block.
        include /etc/nginx/default.d/*.conf;

        location / {
            proxy_pass http://127.0.0.1:8082;
        }

        error_page 404 /404.html;
            location = /40x.html {
        }

        error_page 500 502 503 504 /50x.html;
            location = /50x.html {
        }
    }
}
